JavaScript Closures : 

A closure is a function that remembers variables from its outer function even after the outer function has finished executing.

Closures are used for:

Data hiding
Private variables
Function factories
Callbacks  // function which call itself again and again
Event handlers

1. Basic Closure Example

function outerFunction() {

    let message = "Hello from Closure";

    function innerFunction() {
        console.log(message);
    }

    return innerFunction;
}

let result = outerFunction();

result();


Output

Hello from Closure

How It Works

innerFunction() is inside outerFunction()
It remembers the variable message
Even after outerFunction() completes, message still exists

2. Counter Using Closure

function counter() {

    let count = 0;

    return function () {
        count++;
        console.log("Count:", count);
    };
}

let increment = counter();

increment();
increment();
increment();

Output

Count: 1
Count: 2
Count: 3

function bankAccount() {

    let balance = 1000;

        }
    };
}

account.deposit(500);
account.withdraw(200);

Closures and scoping are core concepts in JavaScript. They help manage variables, memory, data privacy, callbacks, asynchronous programming, and modular code.

1. What is Scope in JavaScript?

Scope determines where variables can be accessed in a program.

JavaScript has:

Global Scope
Function Scope
Block Scope
Lexical Scope

2. Global Scope

Variables declared outside functions are globally accessible.

Variables declared inside a function are accessible only inside that function.

Inner functions can access variables from outer functions.

6. What is Closure?

A closure is created when:

A function remembers variables from its outer scope
Even after the outer function has finished execution

Closures preserve data.

7. Basic Closure Example

function outer() {
    let counter = 0;

    function inner() {
        counter++;
        console.log(counter);
    }

    return inner;
}

const increment = outer();

increment();
increment();
increment();

Output

1
2
3
Explanation

outer() executes once
counter remains in memory
inner() remembers counter

This is a closure.

8. Closure with Data Privacy

Closures can simulate private variables.

function bankAccount(initialBalance) {
    let balance = initialBalance;

    return {
        deposit(amount) {
            balance += amount;
            console.log(`Deposited: ${amount}`);
        },

        withdraw(amount) {
            if (amount <= balance) {
                balance -= amount;
                console.log(`Withdrawn: ${amount}`);
            } else {
                console.log("Insufficient balance");
            }
        },

        getBalance() {
            return balance;
        }
    };
}

account.deposit(500);

(function ($) {

    $.fn.highlightText = function (options) {

        return this.each(function () {

2. Event Delegation

What is Event Delegation?

Event Delegation means:

Attach one event listener to a parent element
Handle events for dynamically added child elements

It uses:

Event Bubbling
Problem Without Delegation
$(".btn").click(function () {
    alert("Button clicked");
});

This works only for existing buttons.

Newly added buttons will NOT work.

Solution Using Event Delegation

$("#container").on("click", ".btn", function () {
    alert("Dynamic Button Clicked");
});

Complete Example

What are Deferred Objects?

Deferred Objects help manage:

Deferred Example
function getData() {

    const deferred = $.Deferred();

    setTimeout(function () {

        let success = true;

        if (success) {
            deferred.resolve("Data Loaded Successfully");
        } else {
            deferred.reject("Error Loading Data");
        }

    }, 2000);

    return deferred.promise();
}

Throttling Example

function throttle(func, limit) {

    let waiting = false;

    return function () {

        if (!waiting) {

            func.apply(this, arguments);

            waiting = true;

            setTimeout(() => {
                waiting = false;
            }, limit);

        }
    };
}
